# Example PySpark cleaning v√† SCD Type 2
from pyspark.sql import functions as F
from delta.tables import DeltaTable

# Data cleaning
df_clean = df_employees \
    .filter(F.col("employee_id").isNotNull()) \
    .withColumn("email", F.lower(F.trim(F.col("email")))) \
    .withColumn("phone", F.regexp_replace(F.col("phone"), "[^0-9]", "")) \
    .dropDuplicates(["employee_id"]) \
    .withColumn("valid_from", F.current_timestamp()) \
    .withColumn("valid_to", F.lit(None).cast("timestamp")) \
    .withColumn("is_current", F.lit(True))

# SCD Type 2 implementation
delta_table = DeltaTable.forPath(spark, "/silver/employees")
delta_table.alias("target") \
    .merge(df_clean.alias("source"), "target.employee_id = source.employee_id AND target.is_current = true") \
    .whenMatchedUpdate(
        condition="target.salary != source.salary OR target.department != source.department",
        set={"is_current": "false", "valid_to": "current_timestamp()"}
    ) \
    .whenNotMatchedInsert(values={
        "employee_id": "source.employee_id",
        "salary": "source.salary",
        "department": "source.department",
        "valid_from": "current_timestamp()",
        "is_current": "true"
    }) \
    .execute()

-- Example Gold layer optimization
CREATE MATERIALIZED VIEW mv_monthly_headcount AS
SELECT
    d.year_number,
    d.month_number,
    dept.department_name,
    pos.job_family,
    COUNT(DISTINCT f.employee_key) as headcount,
    AVG(f.salary_amount) as avg_salary,
    SUM(f.overtime_hours) as total_overtime
FROM fact_employee_monthly f
JOIN dim_date d ON f.date_key = d.date_key
JOIN dim_department dept ON f.department_key = dept.department_key
JOIN dim_position pos ON f.position_key = pos.position_key
WHERE d.day_of_month = 1  -- Month-end snapshot
GROUP BY d.year_number, d.month_number, dept.department_name, pos.job_family;

-- Refresh strategy
REFRESH MATERIALIZED VIEW mv_monthly_headcount;
